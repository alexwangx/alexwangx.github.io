Mac

# Trash replace rm with mv, use trm to run true rm
alias trm='/bin/rm'
trash(){
  mv $@ /Users/alex/.Trash
}
alias rm='trash'

# 修改终端显示方式(centos)
export PS1="[\u@\h \W]\$"

# 设置JAVA_HOME
JAVA_HOME=$(/usr/libexec/java_home -v 1.7)

# 禁止生成"._"文件 http://www.eifr.com/article.php?id=1643&act=print
vi .bashrc
# disable special creation/extraction of ._* files by tar, etc. on Mac OS X
COPYFILE_DISABLE=1;
export COPYFILE_DISABLE


# brew 装机软件
proxychains-ng	wget    python		tmux    jq  unrar

# 清理微信空间
\rm -rf ~/Library/Containers/com.tencent.xinWeChat/Data/Library/Application\ Support/com.tencent.xinWeChat/*



# 关闭摄像头
sudo chmod a-r /System/Library/Frameworks/CoreMediaIO.framework/Versions/A/Resources/VDC.plugin/Contents/MacOS/VDC

#开启摄像头
sudo chmod a+r /System/Library/Frameworks/CoreMediaIO.framework/Versions/A/Resources/VDC.plugin/Contents/MacOS/VDC

# 获取 DNS
networksetup -getdnsservers Wi-Fi

#清理 DNS 缓存
sudo discoveryutil mdnsflushcache

# 设置 DNS
networksetup -setdnsservers Wi-Fi 8.8.8.8 8.8.4.4

AsciiDoc

brew install asciidoctor
brew install pandoc

# 转 html
asciidoctor --backend html5 -a data-uri my_input_file.adoc

# 转 doc
INPUT_ADOC=my_input_file.adoc
asciidoctor --backend docbook --out-file - $INPUT_ADOC|pandoc --from docbook --to docx --output $INPUT_ADOC.docx
