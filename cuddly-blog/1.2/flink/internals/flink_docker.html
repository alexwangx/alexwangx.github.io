Flink Docker安装

环境准备

  1. 安装 Docker

  2. 下载 flink-1.10.2-bin-scala_2.11.tgz

制作 Docker 镜像

tar -zxvf flink-1.10.2-bin-scala_2.11.tgz
vim flink-1.10.2/conf/flink-conf.yaml
jobmanager.rpc.address: flink-jm
taskmanager.numberOfTaskSlots: 10

配置dockerfile

vim dockerfile

FROM centos:centos7
#RUN yum install java-1.8.0-openjdk -y
ADD jdk-8u202-linux-x64.tar.gz /usr/java/jdk
ENV JAVA_HOME /usr/java/jdk/jdk1.8.0_202
ENV PATH ${PATH}:${JAVA_HOME}/bin
ARG FLINK_VERSION=1.10.2
ARG SCALA_VERSION=2.11
ARG FLINK_TAR_NAME=flink-${FLINK_VERSION}-bin-scala_${SCALA_VERSION}.tgz
ENV FLINK_HOME=/flink-${FLINK_VERSION}
ADD ${FLINK_TAR_NAME} /
#RUN mkdir /home
COPY flink_run.sh /home
EXPOSE 8088
VOLUME $FLINK_HOME/conf
RUN chmod u+x /home/flink_run.sh
#ENTRYPOINT cd /home && sh /home/flink_run.sh

ENTRYPOINT ["/home/flink_run.sh"]

配置启动tm, jm

vim flink_run.sh
#!/bin/bash
FLINK_HOME=/flink-1.10.2
FLINK_JOB_MANAGER_SH=$FLINK_HOME/bin/jobmanager.sh
FLINK_TASK_MANAGER_SH=$FLINK_HOME/bin/taskmanager.sh

case "$1" in
"jobmanager")
$FLINK_JOB_MANAGER_SH start-foreground
;;

"taskmanager")
$FLINK_TASK_MANAGER_SH start-foreground
;;

*)
echo "COMMAND ERROR"
;;
esac

build

docker build -f dockerfile -t flink-service:flink1.10.2 .

## 创建Flink的network
docker network create flink


## 启动1个job manager和2个task manager
docker run -itd  --name flink-jm  -p 8081:8081 \
-v $FLINK_HOME/conf:/flink-1.10.2/conf \
-v /tmp/data:/tmp/data \
--network flink flink-service:flink1.10.2 jobmanager

docker run -itd  --name flink-tm1  \
-v $FLINK_HOME/conf:/flink-1.10.2/conf \
-v /tmp/data:/tmp/data \
--network flink flink-service:flink1.10.2 taskmanager

docker run -itd  --name flink-tm2  \
-v $FLINK_HOME/conf:/flink-1.10.2/conf \
-v /tmp/data:/tmp/data \
--network flink flink-service:flink1.10.2 taskmanager


docker start flink-jm flink-tm1 flink-tm2
docker stop flink-jm flink-tm1 flink-tm2

docker ps|grep flink-|awk '{print $1}'|xargs docker stop
docker ps -a |grep flink-|awk '{print $1}'|xargs docker rm

查看是否存在flink-jm和flink-tm容器

docker ps

外部访问:
http://localhost:8081/#/overview

运行示例

echo "127.0.0.1	localhost flink-jm" >> /etc/hosts

# 在 flink 客户端(mac 宿主机) 执行
cd $FLINK_HOME
./bin/flink run ./examples/batch/WordCount.jar

./bin/flink run -m localhost:8081 \
                       ./examples/batch/WordCount.jar \
                       --input file:///tmp/data/README.txt --output file:///tmp/data/wordcount_out

Kafka 安装

# 拉取镜像
docker pull wurstmeister/kafka
docker pull wurstmeister/zookeeper

# 运行 zk
docker run -d --name zookeeper -p 2181:2181 -t wurstmeister/zookeeper

# 运行kafka, 172.30.165.115 为宿主机 IP
docker run --name kafka01 \
-p 9092:9092 \
-e KAFKA_BROKER_ID=0 \
-e KAFKA_ZOOKEEPER_CONNECT=172.30.165.115:2181 \
-e KAFKA_ADVERTISED_LISTENERS=PLAINTEXT://172.30.165.115:9092 \
-e KAFKA_LISTENERS=PLAINTEXT://0.0.0.0:9092 \
-d wurstmeister/kafka
